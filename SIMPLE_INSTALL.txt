📥 METHOD 3: MANUAL (3 Commands)

# Install Node.js
curl -fsSL https://deb.nodesource.com/setup_20.x | sudo -E bash -
sudo apt-get install -y nodejs git

# Download and setup
git clone https://github.com/romerodevv/psgho.git trading-bot
cd trading-bot && npm install

# Configure and run
cp .env.example .env
nano .env  # Edit your private key
node worldchain-trading-bot.js
